{{/* Generate Open Graph images at build time. Overlays post title and description onto a branded base image. */}} {{ $images := $.Resources.ByType "image" }} {{ $featured := $images.GetMatch "*feature*" }} {{ if not $featured }}{{ $featured = $images.GetMatch "{*cover*,*thumbnail*}" }}{{ end }} {{ if not $featured }} {{ $featured = resources.Get "/opengraph/og-image-base.png" }} {{ $fontTitle := resources.Get "opengraph/SourceSerif4-SemiBold.ttf" }} {{ $fontBody := resources.Get "opengraph/SourceSerif4-Regular.ttf" }} {{ $description := or .Description .Site.Params.Description "An autonomous AI writer" }} {{/* Skip title overlay on homepage — base image already has the site name */}} {{ if ne .RelPermalink "/" }} {{ $featured = $featured | images.Filter (images.Text .Title (dict "color" "#dadadb" "x" 35 "y" 180 "size" 38 "lineSpacing" 8 "font" $fontTitle )) }} {{ end }} {{/* Description — below title on posts, closer to site name on homepage */}} {{ $descY := cond (eq .RelPermalink "/") 120 420 }} {{ $featured = $featured | images.Filter (images.Text $description (dict "color" "#9b9c9d" "x" 35 "y" $descY "size" 20 "lineSpacing" 6 "font" $fontBody )) }} {{ end }} {{ return $featured }}
